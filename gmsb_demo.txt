tto:lives
	rdf:type rdf:Property;
	rdfs:label "number of lives"^^xsd:string;
	rdfs:domain tto:Cat;
	rdfs:range xsd:decimal .
	

# Get All Properties of Cat
SELECT DISTINCT ?property ?domain WHERE {
  ?property rdfs:domain ?domain .
  tto:Cat rdfs:subClassOf* ?domain .
}
